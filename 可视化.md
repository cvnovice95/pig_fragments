#### 图像批量显示(show images)
``` python
import matplotlib.pyplot as plt
def plot_imgs(imgs):
    print('imgs length [{}]'.format(len(imgs)))
    row = len(imgs) // 5 + 1
    fig = plt.figure(figsize=(5 * 10, row * 10))  # W,H
    for i, x in enumerate(imgs):
        ax = plt.subplot(row, 5, i + 1)
        plt.imshow(x, cmap=plt.cm.gray)
    plt.show()
# Usage
# img format: [H,W,C]
a = [img1,img2,...]
plot_imgs(a)
```
